// ============================================================ // GM DASHBOARD — Google Apps Script v3 // Handles: financial fields, agenda, action items (ops), // 3-month action plan, narrative/board report fields // // Sheet tabs (auto-created): // "Monthly Data" — one row per month, all financial fields // "Action Log" — action items appended each save (dated) // "Action Plan" — 3-month plan, overwrites each save // "Narrative" — board report narrative, overwrites each save // "Agenda" — agenda items, overwrites each save // ============================================================ var MONTHLY_SHEET = 'Monthly Data'; var ACTIONS_SHEET = 'Action Log'; var PLAN_SHEET = 'Action Plan'; var NARRATIVE_SHEET= 'Narrative'; var AGENDA_SHEET = 'Agenda'; // ============================================================ // ENTRY POINTS // ============================================================ function doGet(e) { var params = e.parameter; if (params.action === 'load') return handleLoad(params.monthKey); return jsonResponse({status:'error', error:'Unknown GET action'}); } function doPost(e) { try { var payload = JSON.parse(e.postData.contents); if (payload.action === 'save') return handleSave(payload); return jsonResponse({status:'error', error:'Unknown POST action'}); } catch(err) { return jsonResponse({status:'error', error:err.toString()}); } } // ============================================================ // LOAD // ============================================================ function handleLoad(monthKey) { var ss = SpreadsheetApp.getActiveSpreadsheet(); // Monthly financial fields var sheet = getOrCreateSheet(ss, MONTHLY_SHEET); var data = sheet.getDataRange().getValues(); var fields = {}, evScope = 'restaurant'; for (var i = 1; i < data.length; i++) { if (data[i][0] === monthKey) { var headers = data[0]; for (var j = 1; j < headers.length; j++) { if (headers[j]) fields[headers[j]] = data[i][j]; } evScope = fields['evScope'] || 'restaurant'; delete fields['evScope']; break; } } return jsonResponse({ status: 'ok', data: { fields: fields, evScope: evScope, actions: loadActionsForMonth(ss, monthKey), actionPlan: loadActionPlan(ss), narrative: loadNarrative(ss), agenda: loadAgenda(ss) } }); } // ============================================================ // SAVE // ============================================================ function handleSave(payload) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var monthKey = payload.monthKey; saveMonthlyData(ss, monthKey, payload.fields||{}, payload.evScope, payload.savedAt); appendActionLog(ss, monthKey, payload.savedAt, payload.actions||[]); saveActionPlan(ss, payload.actionPlan||[]); saveNarrative(ss, payload.narrative||{}, monthKey, payload.savedAt); saveAgenda(ss, payload.agenda||[]); return jsonResponse({status:'ok'}); } // ============================================================ // MONTHLY DATA — financial fields // ============================================================ function saveMonthlyData(ss, monthKey, fields, evScope, savedAt) { var sheet = getOrCreateSheet(ss, MONTHLY_SHEET); var COLUMNS = [ 'monthKey','savedAt','evScope', 'r-food-budget','r-food-actual','r-food-sply', 'r-bev-budget', 'r-bev-actual', 'r-bev-sply', 'b-keg-budget', 'b-keg-actual', 'b-keg-sply', 'b-pack-budget','b-pack-actual','b-pack-sply', 'r-wages','r-labour-target','b-wages','b-labour-target', 'kpi-covers','kpi-turns', 'ev-days-elapsed','ev-days-total', 'meeting-date','meeting-time','meeting-attendees' ]; if (sheet.getLastRow() === 0) { sheet.appendRow(COLUMNS); sheet.getRange(1,1,1,COLUMNS.length).setFontWeight('bold') .setBackground('#1E222D').setFontColor('#C8A84B'); } var row = COLUMNS.map(function(col) { if (col==='monthKey') return monthKey; if (col==='savedAt') return savedAt; if (col==='evScope') return evScope||'restaurant'; return fields[col]!==undefined ? fields[col] : ''; }); var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (data[i][0] === monthKey) { sheet.getRange(i+1,1,1,row.length).setValues([row]); return; } } sheet.appendRow(row); } // ============================================================ // ACTION LOG — ops action items, appended each save // ============================================================ function appendActionLog(ss, monthKey, savedAt, actions) { var sheet = getOrCreateSheet(ss, ACTIONS_SHEET); if (sheet.getLastRow()===0) { var h = ['Month','Saved At','Action','Owner','Due Date','Status']; sheet.appendRow(h); sheet.getRange(1,1,1,h.length).setFontWeight('bold').setBackground('#1E222D').setFontColor('#C8A84B'); } actions.forEach(function(a) { if (!a.action||!a.action.trim()) return; sheet.appendRow([monthKey, savedAt, a.action, a.owner, a.due, a.status]); }); } function loadActionsForMonth(ss, monthKey) { var sheet = getOrCreateSheet(ss, ACTIONS_SHEET); if (sheet.getLastRow()<=1) return []; var data=sheet.getDataRange().getValues(), seen={}, results=[]; for (var i=data.length-1; i>=1; i--) { var row=data[i]; if (row[0]!==monthKey) continue; var key=row[2]; if (!seen[key]) { seen[key]=true; results.unshift({action:row[2],owner:row[3],due:row[4],status:row[5]}); } } return results; } // ============================================================ // 3-MONTH ACTION PLAN — overwrites each save // ============================================================ function saveActionPlan(ss, plan) { var sheet = getOrCreateSheet(ss, PLAN_SHEET); sheet.clearContents(); var h = ['#','Action','Process / Detail','Timing','Responsibility','Last Updated']; sheet.appendRow(h); sheet.getRange(1,1,1,h.length).setFontWeight('bold').setBackground('#1E222D').setFontColor('#C8A84B'); var now = new Date().toISOString(); plan.forEach(function(r,i) { if(!r.action&&!r.process) return; sheet.appendRow([i+1, r.action, r.process, r.timing, r.responsibility, now]); }); } function loadActionPlan(ss) { var sheet = getOrCreateSheet(ss, PLAN_SHEET); if (sheet.getLastRow()<=1) return []; var data=sheet.getRange(2,1,sheet.getLastRow()-1,5).getValues(); return data.map(function(r){ return {action:r[1],process:r[2],timing:r[3],responsibility:r[4]}; }).filter(function(r){ return r.action||r.process; }); } // ============================================================ // NARRATIVE — board report fields, overwrites each save // Also keeps a history tab so previous months are preserved // ============================================================ function saveNarrative(ss, narrative, monthKey, savedAt) { var sheet = getOrCreateSheet(ss, NARRATIVE_SHEET); sheet.clearContents(); var h=['Field','Value','Month','Saved At']; sheet.appendRow(h); sheet.getRange(1,1,1,h.length).setFontWeight('bold').setBackground('#1E222D').setFontColor('#C8A84B'); Object.keys(narrative).forEach(function(key) { sheet.appendRow([key, narrative[key], monthKey, savedAt]); }); } function loadNarrative(ss) { var sheet = getOrCreateSheet(ss, NARRATIVE_SHEET); if (sheet.getLastRow()<=1) return {}; var data=sheet.getRange(2,1,sheet.getLastRow()-1,2).getValues(); var n={}; data.forEach(function(row){ if(row[0]) n[row[0]]=row[1]; }); return n; } // ============================================================ // AGENDA — overwrites each save // ============================================================ function saveAgenda(ss, agenda) { var sheet = getOrCreateSheet(ss, AGENDA_SHEET); sheet.clearContents(); sheet.appendRow(['Agenda Items','Last Updated']); sheet.getRange(1,1,1,2).setFontWeight('bold').setBackground('#1E222D').setFontColor('#C8A84B'); var now=new Date().toISOString(); agenda.forEach(function(item){ sheet.appendRow([item,now]); }); } function loadAgenda(ss) { var sheet=getOrCreateSheet(ss, AGENDA_SHEET); if(sheet.getLastRow()<=1) return []; return sheet.getRange(2,1,sheet.getLastRow()-1,1).getValues() .map(function(r){ return r[0]; }).filter(Boolean); } // ============================================================ // HELPERS // ============================================================ function getOrCreateSheet(ss, name) { var s=ss.getSheetByName(name); if(!s) s=ss.insertSheet(name); return s; } function jsonResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function doOptions(e) { return ContentService.createTextOutput('') .setMimeType(ContentService.MimeType.TEXT); }
